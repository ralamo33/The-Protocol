from textual.app import ComposeResult
from textual.screen import Screen
from textual.widgets import Static
from textual.containers import ScrollableContainer

# Rich markup: [bold cyan]text[/] for cyan, [bold #d4a006]text[/] for amber
_P1 = (
    "You are the [bold cyan]Director[/]. You do not get a title beyond that. "
    "The Foundation does not celebrate its leadership — it consumes them."
)
_P2 = (
    "Anomalies arrive. They are [bold cyan]real[/], they are [bold cyan]dangerous[/], "
    "and they are [bold cyan]already here[/]. Your mandate is containment. "
    "Not understanding. Not elimination. [bold #d4a006]Containment.[/]"
)
_SEP = "─" * 62


class LogScreen(Screen):
    DEFAULT_CSS = """
    LogScreen {
        background: #080c0e;
    }
    #scroll {
        padding: 0 4;
    }
    .header {
        color: #06b6d4;
        padding: 1 0 0 0;
        text-style: bold;
    }
    .sep {
        color: #1e3040;
        margin: 0;
    }
    .briefing {
        color: #c8e8f0;
        padding: 1 0;
    }
    .hint {
        color: #2a4050;
        padding: 1 0 0 0;
    }
    """

    def compose(self) -> ComposeResult:
        with ScrollableContainer(id="scroll"):
            yield Static(
                "THE PROTOCOL  ·  FOUNDATION CONTAINMENT MANAGEMENT SYSTEM",
                classes="header",
            )
            yield Static(_SEP, classes="sep")
            yield Static(_P1, classes="briefing", markup=True)
            yield Static(_P2, classes="briefing", markup=True)
            yield Static(_SEP, classes="sep")
            yield Static("q · quit", classes="hint")

// TECHNICAL DECISIONS — WHY THESE CHOICES
RICH MARKUP
Textual's Static accepts markup=True and passes the string to Rich's parser. [bold cyan]text[/] gives inline color without touching Textual CSS. This is how per-character color will later be applied to Principal-tinted text.
DEFAULT_CSS vs CSS
DEFAULT_CSS on a Screen/Widget is overridable by the App's CSS. Good practice: each widget owns its own default styles, App sets global overrides. Keeps styling co-located with the component that owns it.
ONE STATIC PER PARAGRAPH
Each paragraph is a separate Static widget rather than one big block. Later milestones will animate them in one at a time (typewriter effect) or swap them for reactive LogEntry widgets. Splitting now costs nothing.
